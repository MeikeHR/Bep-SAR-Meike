9    def compute_gini(model):
10       agent_wealths = [agent.wealth for agent in model.schedule.agents]
11       x = sorted(agent_wealths)
12       N = model.num_agents
13       B = sum(xi * (N - i) for i, xi in enumerate(x)) / (N * sum(x))
14       return (1 + (1 / N) - 2 * B)

17   class MoneyModel(Model):
18       """A model with some number of agents."""
19   
20       def __init__(self, N, width, height):
21           self.num_agents = N
22           self.running = True
23           self.grid = MultiGrid(height, width, True)
24           self.schedule = RandomActivation(self)
25           self.datacollector = DataCollector(
26               model_reporters={"Gini": compute_gini},
27               agent_reporters={"Wealth": lambda a: a.wealth}
28           )
29           # Create agents
30           for i in range(self.num_agents):
31               a = MoneyAgent(i, self)
32               self.schedule.add(a)
33               # Add the agent to a random grid cell
34               x = random.randrange(self.grid.width)
35               y = random.randrange(self.grid.height)
36               self.grid.place_agent(a, (x, y))
37   
38       def step(self):
39           self.datacollector.collect(self)
40           self.schedule.step()
41   
42       def run_model(self, n):
43           for i in range(n):
44               self.step()

47   class MoneyAgent(Agent):
48       """ An agent with fixed initial wealth."""
49       def __init__(self, unique_id, model):
50           super().__init__(unique_id, model)
51           self.wealth = 1
52   
53       def move(self):
54           possible_steps = self.model.grid.get_neighborhood(
55               self.pos, moore=True, include_center=False
56           )
57           new_position = random.choice(possible_steps)
58           self.model.grid.move_agent(self, new_position)
59   
60       def give_money(self):
61           cellmates = self.model.grid.get_cell_list_contents([self.pos])
62           if len(cellmates) > 1:
63               other = random.choice(cellmates)
64               other.wealth += 1
65               self.wealth -= 1
66   
67       def step(self):
68           self.move()
69           if self.wealth > 0:
70               self.give_money()
